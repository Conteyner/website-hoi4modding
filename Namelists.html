Названия дивизий

Названия подразделений определяются в /Hearts of Iron IV/common/units/names_divisions в текстовых файлах для конкретной страны. Имя файла не имеет значения, но обычно он имеет следующий формат TAG_names_divisions.txt.

После создания файла начните со списка имен, установленного в вашем OOB, который находится в /Hearts of Iron IV/history/units, например, NEP_MTN_01. Вы должны следовать следующему форматированию 

NEP_MTN_01 = { # DIVISIONS_NAME_GROUP, установленный в папке ваших юнитов
	name = "Mountain Divisions" # Имя, которое вы хотите использовать для списка имен, может быть строкой или заключено в кавычки.
	for_countries = { NEP } # Устанавливает, какие страны могут использовать этот список имен, измените NEP на тег вашей страны.
	can_use = { always = yes } # Триггер страны, который может заблокировать/разблокировать группу имен при определенных условиях
	division_types = { "mountaineers" } # Устанавливает, какие типы юнитов могут использовать список, замените на любой тип юнита.
	fallback_name = "%d Nepali Dibhijana" # Он будет использоваться, если у вас закончились нумерованные деления. 
                                                #Всегда используйте либо %d для десятичных чисел, либо %s для римских цифр, 
                                                #но вы  можете изменить остальные значения.
	link_numbering_with = { NEP_INF_01 } # Альтернатива запасному имени, если у вас закончились номера в этой группе,
                                            #вы можете использовать другой список после нее.
	ordered = {
		1 = { "%d Nepali Dibhijana" } # Первая дивизия.
		2 = { "%d Nepali Dibhijana" } # Числа всегда должны начинаться с 1 и идти вверх
		4 = { "%d Nepali Dibhijana" } # Номера не обязательно должны быть последовательными
	}
}

Названия Флота

Названия кораблей определяются в /Hearts of Iron IV/common/units/names_ships в текстовых файлах для конкретной страны.
Имя файла не имеет значения, но обычно он имеет следующий формат TAG_ship_names.txt. В этом файле название не зависит от установленного имени в вашем OOB-файле, как в названиях подразделений, вместо этого это любой корабль. Вы должны следовать следующему форматированию 

PAK_DD_HISTORICAL = { # Любое имя по вашему желанию
	name = NAME_THEME_HISTORICAL_DESTROYERS # Имя, которое отображается в морском дизайнере, может быть строкой или в кавычках.
	for_countries = { PAK } # Устанавливает, какие страны могут использовать этот список имен, измените PAK на тег вашей страны
	type = ship # Не меняйте
	ship_types = { ship_hull_light destroyer } # Первый - технология MTG, второй - не MTG. Установите на нужную технологию
	prefix = "PNS " # Имя, которое появляется перед вашими кораблями
	fallback_Name = "Destroyer %d" # Это будет использоваться, если у вас закончатся пронумерованные деления. 
    #Всегда используйте либо %d для десятичных чисел, либо %s для римских цифр, но остальное можно изменить
	unique = {
		"Shamsher" "Tippu Sultan" "Tariq" # Перечислите названия кораблей, которые могут использовать, не используйте запятые
	}
}

Общие названия

Определены в /Hearts of Iron IV/common/units/names/*.txt, для стран, у которых нет конкретных названий для типов снаряжения, особенно для пользовательского снаряжения, которое не каждая нация будет использовать. 

generic = { # Применяет его ко всем странам
	submarine = { # Название оборудования
		prefix = "" # Название, которое идет перед оборудованием
		generic = { "Submarine" } # Устанавливает название оборудования
		unique = {  } # Для типового оборудования ничего не указывайте здесь
	}
}

Оперативные кодовые названия

Хранящиеся в /Hearts of Iron IV/common/units/codenames_operatives, эти кодовые имена будут случайным образом присвоены оперативникам указанных стран. Пример определения выглядит следующим образом 

codename_list_id = {				# Идентификатор namelist
	name = codename_list_name		# Имя списка имен, может совпадать с ID

	for_countries = { TAG1 TAG2 }	# Страны, использующие его

	type = codename					# Чтобы сообщить игре, что это список кодовых имен

	fallback_name = "Agent %d"		# В случае, если уникальность закончится.

	unique = {						# Уникальные кодовые имена, 1 оперативник - 1 кодовое имя
		"Codename 1"
		"Codename 2"
	}
}
